#include <iostream>
#include<thread>
#include<mutex>
#include<condition_variable>

int num =1;
const int NUM_MAX = 10;
bool is_num_even = false;

std::mutex mtx;
std::condition_variable cv;

void printOdd() {
    while(true) {
        std::unique_lock<std::mutex> lock(mtx);
        cv.wait(lock,[] {return !is_num_even || num>NUM_MAX;});
        
        if(num>NUM_MAX) break;
        
        std::cout<<"Odd:"<<num<<std::endl;
        ++num;
        is_num_even =true;
        cv.notify_all();
        
    }
}

void printEven() {
    while(true) {
        std::unique_lock<std::mutex> lock(mtx);
        cv.wait(lock,[] {return is_num_even || num>NUM_MAX;});
        
        if(num>NUM_MAX) break;
        
        std::cout<<"Even:"<<num<<std::endl;
        ++num;
        is_num_even =false;
        cv.notify_all();
        
    }
}


int main()
{
    std::thread t2(printOdd);
    std::thread t1(printEven);
   

    t1.join();
    t2.join();

    return 0;
}
